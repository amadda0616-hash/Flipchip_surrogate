import pandas as pd
import numpy as np
from sklearn.mixture import GaussianMixture
import seaborn as sns
import matplotlib.pyplot as plt

# 1. GMM 모델 학습 (sdv 대신 사용 권장)
# n_components를 조절하여 원본과 가장 비슷한 부드러운 곡선을 찾으세요.
gmm = GaussianMixture(n_components=10, covariance_type='full', random_state=42)
gmm.fit(df)  # df는 원본 데이터

# 2. 증강 데이터 생성
n_samples = len(df) * 2
augmented_data, _ = gmm.sample(n_samples)
df_augmented = pd.DataFrame(augmented_data, columns=df.columns)

# 3. 분포 비교 확인
plt.figure(figsize=(10, 5))
sns.kdeplot(df['주요변수'], label='Original', color='blue')
sns.kdeplot(df_augmented['주요변수'], label='Augmented', color='orange')
plt.legend()
plt.show()
